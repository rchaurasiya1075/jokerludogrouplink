<div id="ascii-container">Loading animation...</div>

<script>
  const frameCount = 5; // üëà Change this to match your total number of frames
  let currentFrame = 0;
  let frames = [];
  const container = document.getElementById('ascii-container');

  async function loadFrames() {
    for (let i = 1; i <= frameCount; i++) {
      try {
        const res = await fetch(`frames/frame${i}.txt`);
        const text = await res.text();
        frames.push(text);
      } catch (err) {
        frames.push(`(Frame ${i} missing)`);
      }
    }
  }

  function playAnimation() {
    setInterval(() => {
      container.textContent = frames[currentFrame];
      currentFrame = (currentFrame + 1) % frames.length;
    }, 200); // ‚è± Change speed here (in milliseconds)
  }

  loadFrames().then(playAnimation);
</script>
